\end{aligned}\\
\begin{aligned}s=\dfrac{1}{2}\left( v_{0}+v_{t}\right) t\\
s=v_{0}t+\dfrac{1}{2}at^{2}\\
v_{t}^{2}-v_{0}^{2}=2as\end{aligned}\\
V_{instant}=V_\dfrac{t}{2}=\overline{V}=\dfrac{V_{0}+V_{t}}{2}\left( Midpoint of time\right) \\
